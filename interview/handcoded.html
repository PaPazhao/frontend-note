    # 实现 new

    /**
 * new 方法的实现
 * @param {构造函数} constructor
 * @param  {构造函数的参数} args
 */
function newFunction(constructor, ...args) {
  const obj = Object.create(constructor.prototype);
  // 2. 将 obj 作为 this ，执行 constructor ，传入参数
  constructor.apply(obj, args);
  // 3. 返回 obj
  return obj;
}

    # 防抖

    function debounce(fn, delay = 200) {
  let timer = 0;

  return function (...args) {
    if (timer) {
      clearTimeout(timer);
    }

    timer = setTimeout(() => {
      timer = 0;
      return fn.apply(this, args);
    }, delay);
  };
}

    # 节流

    function throttle(fn, delay = 200) {
  let timer = 0;

  return function (...args) {
    if (timer) {
      return;
    }

    timer = setTimeout(() => {
      timer = 0;
      return fn.apply(this, args);
    }, delay);
  };
}

    # 深拷贝

    /**
 * 深拷贝
 * @param {要复制的对象} target
 */
function deepCopy(target, map = new WeakMap()) {
  const targetType = typeof target;
  if (target === null || targetType !== "object") {
    return target;
  }

  // map 保存了已经 copy [old: new] 的对象,防止递归 copy
  const objFromWeakMap = map.get(target);
  if (objFromWeakMap) {
    return objFromWeakMap;
  }

  let newObj = {};
  map.set(target, newObj);

  if (Array.isArray(target)) {
    newObj = target.map((item) => deepCopy(item, map));

    return newObj;
  }
  if (target instanceof Map) {
    const newMap = new Map();
    target.forEach((v, k) => {
      const value = deepCopy(v, map);
      const key = deepCopy(k, map);
      newMap.set(key, value);
    });
    return newMap;
  }
  if (target instanceof Set) {
    const newSet = new Set();
    target.forEach((v) => {
      const value = deepCopy(v, map);
      newSet.add(value);
    });
    return newSet;
  }

  for (const key in target) {
    const value = target[key];
    const v = deepCopy(value, map);
    newObj[key] = v;
  }

  const symbolKeys = Object.getOwnPropertySymbols(target);
  symbolKeys.forEach((symbol) => {
    const newSymbol = Symbol(symbol.description);
    newObj[newSymbol] = deepCopy(target[symbol]);
  });

  return newObj;
}

    /**
 * 获取类型
 * @param {any} x
 * @returns x 的类型
 */
function getTypes(x) {
  // [object String]
  const originType = Object.prototype.toString.call(x);
  const spaceIndex = originType.indexOf(" ");
  return originType.slice(spaceIndex + 1, -1).toLowerCase();
}

    # 手写继承

    function createObject(o) {
  function F() {}
  F.prototype = o;
  return new F();
}

function inherit(SubType, SuperType = Object) {
  SubType.prototype = createObject(SuperType.prototype);
  Object.defineProperty(SubType.prototype, "constructor", {
    configurable: true,
    enumerable: false,
    writable: true,
    value: SubType,
  });
}

function Person(name) {
  this.name = name;
}
function Student(name, score) {
  Person.call(this, name);
  this.score = score;
}
inherit(Student, Person);

    # 图片懒加载

    图片懒加载原理:

    • 添加图片的时候 src 先设置一个占位图,真正的图片地址设置在 data-src 属性
    <img src="./img/loading.gif" data-src="./img/animal1.jpeg">
    • 监听滚动, 设置一个回调事件 mapImageAndTryload 回调事件里处理懒加载的逻辑, 注意调用回调要节流
    <script src="https://cdn.bootcdn.net/ajax/libs/lodash.js/4.17.21/lodash.min.js"></script>

    window.addEventListener("scroll", _.throttle(mapImageAndTryload, 100));
    • 选中带 data-src 属性的 img, 然后来判断图片是否出现在 window 的可视区域. 判断的依据是获取图片的 top 值跟 window.innerHeight 比较 img.rect.top < window.innerHeight 则图片就在可视区域内了. 然后把图片的 data-src 的值赋值给 src 属性并且赋值后删除 data-src 属性
    function mapImageAndTryload() {
  const images = document.querySelectorAll("img[data-src]");
  if (images.length === 0) {
    return;
  }

  images.forEach((image) => {
    const rect = image.getBoundingClientRect();
    if (rect.top < window.innerHeight) {
      image.src = image.dataset.src;
      image.removeAttribute("data-src");
    }
  });
}

    响应式原理

    整条说响应式响应式,那么到底啥是响应式呢?

    举个例子, 例如你有个购物车程序, 你买了一本书经典书 <你不知道的 JavaScript>

    let book = {
  name: '你不知道的 JavaScript',
  price: 100,
  quantity: 1
}

let total = book.price * book.quantity
console.log(`总价格: ${total}`)

    我们得到的结果是:

    总价格: 100

    购物车一般都有修改数量的按钮,当我们修改书的数量变成 2

    // 修改数量
book.quantity = 2 
console.log(`总价格: ${total}`)

    如果是响应式的话,我们得到的结果就是

    总价格: 200

    然而实际结果是

    总价格: 100

    所以说响应式到底是什么. 响应式就是先保存某一段代码逻辑,监听某个属性的变化的时候再把这段代码执行一遍. 在这里例子中这段逻辑就是

    total = book.price * book.quantity

    监听 book.quantity 的变化,然后把这段计算总价格的逻辑再执行一遍.

    我们的解决方案就是把这段代码保存到一个函数里面, 当需要的时候调用一下. 依赖某个属性变化而需要重新计算的逻辑可能有多个,所以我们把他保存到一个数组里面,当属性变化的时候把这个数组的所有函数都执行一遍

    let book = {
  name: '你不知道的 JavaScript',
  price: 100,
  quantity: 1
}
let total = 0
let storage = []
let target = null
target = () => { total = book.price * book.quantity }

target() 
console.log('总价格: ', total) // => 100

book.quantity = 2
target()
console.log('总价格: ', total) / => 200

    依赖 quantity 属性的变化而需要进行调用的代码也许有多段,那么我们需要一个容器来保存, 当属性发生变化的时候保存的所有代码都去执行一遍.

    let storage = []

// 记录
function record() {
  storage.push(target)
}

// 执行
function replay() {
  storage.forEach(fn => fn())
}

    那么就是这样

    let book = {
  name: '你不知道的 JavaScript',
  price: 100,
  quantity: 1
}
let total = 0
let target = null
target = () => { total = book.price * book.quantity }

let storage = []
function record() {
  storage.push(target)
}


target()
record()
console.log('总价格: ', total) // => 总价格:  100

book.quantity = 2
target()
console.log('总价格: ', total) // => 总价格:  200

    上面的代码就是非常简单的实现了响应式的手动挡版本. 而且无法复用需要进行封装.

    我们需要一个类 Dep 来管理依赖关系, 当需要创建依赖关系的时候就实例化一个 Dep 实例,然后添加依赖, 当需要运行的时候再去执行

    class Dep {
  constructor() {
    this.subscribers = []
  }
  
  depend() {
    if(target && !this.subscribers.includes(target)) {
      this.subscribers.push(target)
    }
  }

  notify() {
    this.subscribers.forEach(fn => fn())
  }
}

    然后用 dep 来替代之前的依赖管理

    let book = {
  name: '你不知道的 JavaScript',
  price: 100,
  quantity: 1
}
let total = 0
let target = () => { total = book.price * book.quantity }
 
const dep = new Dep()
dep.depend()

target()
console.log('总价格: ', total) //=> 总价格:  100

book.quantity = 2
dep.notify()
console.log('总价格: ', total) //=> 总价格:  200

    运行后得到了期望的结果. 现在依赖的这些函数是手动进行管理的,我们需要多个的时候就不方便了.封装到 watcher 函数里面

    
function watcher(fn) {
  target = fn
  dep.depend()
  fn()
  target = null
}

    这样我们把 target 的匿名函数封装到了 watcher 里面, 并且顺便建立依赖关系

    let book = {
  name: '你不知道的 JavaScript',
  price: 100,
  quantity: 1
}
let total = 0
let target = null 

const dep = new Dep()

watcher(() => { 
  total = book.price * book.quantity 
}) 
console.log('总价格: ', total) //=> 总价格:  100

book.quantity = 2
dep.notify()
console.log('总价格: ', total) //=> 总价格:  200

    现在的代码更加自动化了一些,不过目前仍然是手动挡,而且只能手动指定一个依赖. 我们的目标是自动收集依赖,自动去执行.

    对于这个自动收集的方案有两种,一种是使用 Object.defineProperty 的 get, set 方法, get 的时候收集依赖, set 的时候触发依赖更新. 另一种是 ES6 的 Proxy

    let internalValue = book.quantity
Object.defineProperty(book, 'quantity', {
  get() {
    console.log('收集依赖')
    return internalValue
  },
  set(newValue) {
    internalValue = newValue
    console.log('执行更新')
  }
})

    这样当你读写 book.quantity 的时候相应的就会去执行 get/set 方法,同时就会执行我们添加的依赖收集和更新的逻辑,这一切就是自动的. 手动挡变自动挡的核心就是这里了.

    我们如果需要对某个对象监听多个属性,那么就需要对这个对象的属性进行遍历,然后向上面那样增加依赖

    function reactive(obj) {
  Object.keys(obj).forEach(key => {
    let value = obj[key]
    const dep = new Dep()

    Object.defineProperty(obj, key, {
      get() { 
        dep.depend()
        return value
      },
      set(newValue) {
        value = newValue 
        dep.notify()
      }
    })
  })

  return obj
}

    这样依赖的收集和更新都变成自动化, 但是目前为每个对象的每个 key 都定义了一个 dep, 可以用一个map表来管理依赖

    const depsmap = new WeakMap()
function getDep(obj, key) {
  let depmap = depsmap.get(obj)
  if(!depmap) {
    depmap = new Map()
    depsmap.set(obj, depmap)
  }
  let dep = depmap.get(key)
  if(!dep) {
    dep = new Dep() 
    depmap.set(key, dep)
  }
  return dep
}

  constructor() { 
    this.subscribers = []
  }

  depend() {
    if (target && !this.subscribers.includes(target)) {
      this.subscribers.push(target)
    }
  }

  notify() {
    this.subscribers.forEach(fn => fn())
  }
}

const depsmap = new WeakMap()
function getDep(obj, key) {
  let depmap = depsmap.get(obj)
  if(!depmap) {
    depmap = new Map()
    depsmap.set(obj, depmap)
  }
  let dep = depmap.get(key)
  if(!dep) {
    dep = new Dep() 
    depmap.set(key, dep)
  }
  return dep
}
function reactive(obj) {
  Object.keys(obj).forEach(key => {
    let value = obj[key]
     
    Object.defineProperty(obj, key, {
      get() {
        const dep = getDep(obj, key)
        dep.depend()
        return value
      },
      set(newValue) {
        value = newValue
        const dep = getDep(obj, key)
        dep.notify()
      }
    })
  })

  return obj
}
